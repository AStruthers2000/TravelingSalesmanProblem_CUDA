Traveling Salesman Problem
The primary objective of this project is to explore the potential of Ant Colony Optimization (ACO), a metaheuristic inspired by the foraging behavior of ants, in solving the Travelling Salesman Problem (TSP) using Compute Unified Device Architecture (CUDA). The goal of this project is to demonstrate the feasibility and advantages of using GPGPU-based parallel computing in solving complex combinatorial optimization problems. By harnessing the parallel processing capabilities of CUDA, we seek to develop an efficient parallel implementation of ACO for TSP solving on GPGPUs, aiming to significantly reduce the computational time required to find near-optimal solutions for large-scale instances of the problem.

75int main(int argc, char** argv)
76{
77 string path = R"(.\Datasets\)";
78
79 //track the upper and lower bounds on the problem instances we want to solve
80 int lower_size = 0;
81 int upper_size = numeric_limits<int>::max();
82
83 //if 3 arguments were passed (exe name and 2 numbers), read them as lower and upper limits
84 if(argc >= 3)
85 {
86 lower_size = stoi(argv[1]);
87 upper_size = stoi(argv[2]);
88 }
89 //if only 2 arguments were passed, read the 2nd as the lower limit
90 else if(argc == 2)
91 {
92 lower_size = stoi(argv[1]);
93 }
94
95 namespace fs = std::filesystem;
96
97 //regex pattern to read the numbers in the filename, useful for finding the problem size before parsing the data
98 regex size_regex("\\d+");
99 for (const auto& entry : fs::directory_iterator(path))
100 {
101 // entry.path() returns the full path of the file
102 const string file = entry.path().filename().generic_string();
103 const string stripped_file = split_string_by_token(file, '.')[0];
104 smatch match;
105
106 if (regex_search(file, match, size_regex))
107 {
108 //read the part of the filename that is only numbers as an int
109 int size = stoi(match.str());
110
111 //skip this problem if its size is outside the problem bounds
112 if(size < lower_size || size > upper_size) continue;
113
114 //load the adjacency matrix into memory as a flattened 1D array
115 auto adj_mat = (double *) malloc(size * size * sizeof(double));
116 load_adjacency_matrix_from_file(entry.path().generic_string(), size, adj_mat);
117
118 cout << "Starting experimentation on problem " << stripped_file << endl;
119
120 int experiments = 30;
121 double sols = 0.0;
122 double min = numeric_limits<double>::max();
123 auto start = high_resolution_clock::now();
124
125 //perform experiments number of independent experiments on the given problem
126 for (int i = 0; i < experiments; i++)
127 {
128 auto problem_start = high_resolution_clock::now();
129
130 //call the ACO solver with the problem definition (adjacency matrix) and problem size
131 double sol = ACO_main(adj_mat, size);
132
133 auto problem_end = high_resolution_clock::now();
134 auto problem_duration = duration_cast<microseconds>(problem_end - problem_start).count();
135
136 write_to_file(stripped_file, i, sol, static_cast<double>(problem_duration) / 1e06);
137
138 sols += sol;
139 if (sol < min)
140 {
141 min = sol;
142 }
143 }
144
145 auto end = high_resolution_clock::now();
146 auto duration = duration_cast<microseconds>(end - start).count();
147 double avg = sols / experiments;
148
149 printf("Best solution for %d experiments: \t%2.4f\n", experiments, min);
150 printf("Average solution for %d experiments: \t%2.4f\n", experiments, avg);
151 printf("Took %lld seconds for all %d experiments\n\tAverage of %lld milliseconds per experiment\n",
152 duration / static_cast<long long>(1e06), experiments, (duration / 1000) / experiments);
153 printf("=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=\n");
154
155 //free our problem. If we malloc memory, we also need to free memory!!!!
156 free(adj_mat);
157 }
158 }
159 return EXIT_SUCCESS;
160}
